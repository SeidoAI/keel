# Keel v0.6b — Workspace Primitive Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the workspace primitive from the v0.6 design — a centralized synchronisation layer for shared concept nodes across projects, with git-aligned sync semantics (pull/push/promote/fork/copy), bidirectional project↔workspace registry, agent-mediated 3-way merges via structured merge briefs, and two new PM slash commands (`/pm-project-create`, `/pm-project-sync`). Migrates the three test projects into a `seido-workspace`.

**Architecture:** Multi-repo — workspace is its own git repo, projects stay independent. Node files in projects carry `origin`/`scope`/`workspace_sha` frontmatter distinguishing workspace-origin from local. Sync via command-driven pull/push operations that mirror git semantics. Conflicts surface as structured merge briefs for agent mediation rather than text conflict markers. Workspace-aware lint rules extend Plan 1's framework.

**Depends on:** Plan 1 (v0.6a) — uses its lint framework (`@register_rule`, stages), manifest ownership fields (`produced_by`/`owned_by`), and `keel session derive-branch`.

**Tech Stack:** Python 3.10+, Pydantic v2, Click, Jinja2, pytest. Git operations via existing subprocess patterns (see `key_allocator.py` for the existing `project_lock`). Reuses `paths.py`, `locks.py`, validator rule registration from v0.5.

```python
"""Workspace model (v0.6b).

A workspace is a centralized shared-concepts layer for N projects.
workspace.yaml lives at the workspace repo root and maintains the
bidirectional registry of projects with their last-sync state.
"""

from __future__ import annotations

from datetime import datetime
from typing import Literal
from uuid import UUID

from pydantic import BaseModel, ConfigDict, Field, model_validator

SUPPORTED_SCHEMA_VERSIONS: tuple[int, ...] = (1,)


class WorkspaceProjectEntry(BaseModel):
    """One member project in a workspace's registry."""

    model_config = ConfigDict(extra="forbid")

    slug: str  # workspace-local alias, e.g. "kbp"
    name: str  # matches project.yaml.name
    path: str  # relative (to workspace root) or absolute path

    last_pulled_sha: str | None = None
    last_pulled_at: datetime | None = None
    last_pushed_sha: str | None = None
    last_pushed_at: datetime | None = None


class Workspace(BaseModel):
    """workspace.yaml root."""

    model_config = ConfigDict(extra="forbid")

    uuid: UUID
    name: str
    slug: str
    description: str
    schema_version: int
    keel_version: str
    created_at: datetime
    updated_at: datetime
    projects: list[WorkspaceProjectEntry] = Field(default_factory=list)

    @model_validator(mode="after")
    def _schema_version_supported(self) -> "Workspace":
        if self.schema_version not in SUPPORTED_SCHEMA_VERSIONS:
            raise ValueError(
                f"workspace schema_version {self.schema_version} not supported "
                f"(this keel supports {SUPPORTED_SCHEMA_VERSIONS})"
            )
        return self

    @model_validator(mode="after")
    def _project_slugs_unique(self) -> "Workspace":
        slugs = [p.slug for p in self.projects]
        if len(slugs) != len(set(slugs)):
            dups = {s for s in slugs if slugs.count(s) > 1}
            raise ValueError(f"duplicate project slugs in workspace: {dups}")
        return self
```

```python
"""Read/write workspace.yaml.

Follows the same frontmatter-YAML pattern as other keel stores.
Registry-mutation functions acquire a lock on the workspace's lock file
and update atomically.
"""

from __future__ import annotations

from datetime import datetime, timezone
from pathlib import Path

from keel.core.locks import project_lock
from keel.core.parser import dump_frontmatter_yaml, load_frontmatter_yaml
from keel.core.paths import workspace_yaml_path
from keel.models.workspace import Workspace, WorkspaceProjectEntry


def workspace_exists(workspace_dir: Path) -> bool:
    return workspace_yaml_path(workspace_dir).exists()


def load_workspace(workspace_dir: Path) -> Workspace:
    path = workspace_yaml_path(workspace_dir)
    frontmatter, _body = load_frontmatter_yaml(path)
    return Workspace.model_validate(frontmatter)


def save_workspace(workspace_dir: Path, workspace: Workspace) -> None:
    path = workspace_yaml_path(workspace_dir)
    path.parent.mkdir(parents=True, exist_ok=True)
    # Bump updated_at on every save.
    workspace = workspace.model_copy(update={"updated_at": datetime.now(tz=timezone.utc)})
    dump_frontmatter_yaml(
        path,
        frontmatter=workspace.model_dump(mode="json", exclude_none=True),
        body="",
    )


def add_project(workspace_dir: Path, entry: WorkspaceProjectEntry) -> None:
    with project_lock(workspace_dir):
        ws = load_workspace(workspace_dir)
        if any(p.slug == entry.slug for p in ws.projects):
            raise ValueError(f"project slug '{entry.slug}' already registered")
        ws = ws.model_copy(update={"projects": [*ws.projects, entry]})
        save_workspace(workspace_dir, ws)


def remove_project(workspace_dir: Path, *, slug: str) -> None:
    with project_lock(workspace_dir):
        ws = load_workspace(workspace_dir)
        kept = [p for p in ws.projects if p.slug != slug]
        if len(kept) == len(ws.projects):
            raise ValueError(f"project slug '{slug}' not found in workspace")
        ws = ws.model_copy(update={"projects": kept})
        save_workspace(workspace_dir, ws)


def update_project_pull_state(
    workspace_dir: Path, *, slug: str, sha: str, at: datetime
) -> None:
    with project_lock(workspace_dir):
        ws = load_workspace(workspace_dir)
        updated = [
            p.model_copy(update={"last_pulled_sha": sha, "last_pulled_at": at})
            if p.slug == slug else p
            for p in ws.projects
        ]
        ws = ws.model_copy(update={"projects": updated})
        save_workspace(workspace_dir, ws)


def update_project_push_state(
    workspace_dir: Path, *, slug: str, sha: str, at: datetime
) -> None:
    with project_lock(workspace_dir):
        ws = load_workspace(workspace_dir)
        updated = [
            p.model_copy(update={"last_pushed_sha": sha, "last_pushed_at": at})
            if p.slug == slug else p
            for p in ws.projects
        ]
        ws = ws.model_copy(update={"projects": updated})
        save_workspace(workspace_dir, ws)
```

@click.option("--name", required=True, help="Human-readable workspace name.")
@click.option("--slug", required=True, help="Short alias used in CLI output.")
@click.option("--description", default="", help="One-liner.")
@click.option(

    type=click.Path(path_type=Path, file_okay=False, dir_okay=True),
    default=".", show_default=True,

def workspace_init_cmd(
    name: str, slug: str, description: str, workspace_dir: Path
) -> None:
    """Bootstrap a new workspace in workspace-dir."""
    resolved = workspace_dir.expanduser().resolve()
    resolved.mkdir(parents=True, exist_ok=True)
    if workspace_exists(resolved):
        raise click.ClickException(
            f"workspace already exists at {resolved} "
            "(workspace.yaml present)"
        )

    ws = Workspace(
        uuid=uuid4(),
        name=name, slug=slug, description=description,
        schema_version=1, keel_version=KEEL_VERSION,
        created_at=now, updated_at=now,
    )

    workspace_nodes_dir(resolved).mkdir(parents=True, exist_ok=True)

    if not (resolved / ".git").exists():

    click.echo(f"✓ Workspace '{name}' initialized at {resolved}")

@workspace_cmd.command("link")
@click.argument("workspace_path",
                type=click.Path(path_type=Path, file_okay=False, dir_okay=True))
@click.option(
    "--project-dir",
    type=click.Path(path_type=Path, file_okay=False, dir_okay=True),
    default=".", show_default=True,
)
@click.option("--slug", required=True, help="Workspace-local alias for this project.")
def workspace_link_cmd(
    workspace_path: Path, project_dir: Path, slug: str
) -> None:
    """Register this project with a workspace (bidirectional)."""
    proj_resolved = project_dir.expanduser().resolve()
    ws_resolved = workspace_path.expanduser().resolve()
    _require_project(proj_resolved)

    if not workspace_exists(ws_resolved):
        raise click.ClickException(
            f"no workspace.yaml at {ws_resolved}"
        )

    # Reject if already linked.
    cfg = load_project_config(proj_resolved)
    if cfg.workspace is not None:
        raise click.ClickException(
            f"project is already linked to workspace at {cfg.workspace.path}; "
            "run `keel workspace unlink` first"
        )

    # Project-side: write workspace pointer with a relative path.
    try:
        rel = ws_resolved.relative_to(proj_resolved.parent)
        pointer_path = f"../{rel.name}" if rel.parts[0] == ".." or str(rel).startswith("..") else str(rel)
    except ValueError:
        pointer_path = str(ws_resolved)  # absolute fallback

    # Use a simpler relative resolution: compute path from project to workspace.
    import os
    try:
        pointer_path = os.path.relpath(ws_resolved, proj_resolved)
    except ValueError:
        pointer_path = str(ws_resolved)

    cfg_new = cfg.model_copy(update={
        "workspace": ProjectWorkspacePointer(path=pointer_path),
    })
    save_project_config(proj_resolved, cfg_new)

    # Workspace-side: add project entry.
    ws_relative_back = os.path.relpath(proj_resolved, ws_resolved)
    add_project(ws_resolved, WorkspaceProjectEntry(
        slug=slug, name=cfg.name, path=ws_relative_back,
    ))

    click.echo(f"✓ Linked {cfg.name} ↔ workspace {load_workspace(ws_resolved).slug}")
    click.echo(f"  project.yaml.workspace.path: {pointer_path}")
    click.echo(f"  workspace.yaml.projects[{slug}].path: {ws_relative_back}")

@workspace_cmd.command("unlink")
@click.option(
    "--project-dir",
    type=click.Path(path_type=Path, file_okay=False, dir_okay=True),
    default=".", show_default=True,
)
@click.option("--force", is_flag=True, default=False,
              help="Force remove project-side pointer even if workspace missing.")
def workspace_unlink_cmd(project_dir: Path, force: bool) -> None:
    """Unlink this project from its workspace."""
    proj = project_dir.expanduser().resolve()
    _require_project(proj)
    cfg = load_project_config(proj)
    if cfg.workspace is None:
        raise click.ClickException("project is not linked to any workspace")

    ws_path_raw = cfg.workspace.path
    ws_resolved = (proj / ws_path_raw).resolve()

    if not workspace_exists(ws_resolved):
        if not force:
            raise click.ClickException(
                f"workspace at {ws_resolved} not found; re-run with --force "
                "to remove the project-side pointer only"
            )
    else:
        # Workspace-side: remove project entry if present.
        ws = load_workspace(ws_resolved)
        matching = [p for p in ws.projects if _paths_point_to_same(
            ws_resolved, p.path, proj
        )]
        for p in matching:
            try:
                from keel.core.workspace_store import remove_project
                remove_project(ws_resolved, slug=p.slug)
            except ValueError:
                pass  # already gone

    # Project-side: clear pointer.
    cfg_new = cfg.model_copy(update={"workspace": None})
    save_project_config(proj, cfg_new)

    click.echo("✓ Unlinked from workspace.")


def _paths_point_to_same(ws_dir: Path, stored_relative: str, project_dir: Path) -> bool:
    """Workspace-side entry path is relative to workspace root; resolve and compare."""
    resolved = (ws_dir / stored_relative).resolve()
    return resolved == project_dir

@workspace_cmd.command("status")
@click.option(
    "--workspace-dir",
    type=click.Path(path_type=Path, file_okay=False, dir_okay=True),
    default=None,
)
@click.option(
    "--project-dir",

    default=None,
)
@click.option(
    "--format", "output_format",
    type=click.Choice(["text", "json"]), default="text", show_default=True,
)
def workspace_status_cmd(
    workspace_dir: Path | None, project_dir: Path | None, output_format: str
) -> None:
    """Show sync state.

    From workspace-dir: cross-project summary.
    From project-dir: per-node drift (behind / ahead of workspace).
    If neither given, try cwd (workspace first, then project).
    """
    resolved_ws = workspace_dir.expanduser().resolve() if workspace_dir else None
    resolved_proj = project_dir.expanduser().resolve() if project_dir else None

    if not resolved_ws and not resolved_proj:
        cwd = Path(".").resolve()
        if workspace_exists(cwd):
            resolved_ws = cwd
        else:
            resolved_proj = cwd
            _require_project(resolved_proj)

    if resolved_ws:
        _status_workspace(resolved_ws, output_format)
    elif resolved_proj:
        _status_project(resolved_proj, output_format)


def _status_workspace(ws_dir: Path, output_format: str) -> None:
    ws = load_workspace(ws_dir)
    rows = []
    for p in ws.projects:
        rows.append({
            "slug": p.slug, "name": p.name,
            "last_pulled_at": p.last_pulled_at.isoformat() if p.last_pulled_at else None,
            "last_pushed_at": p.last_pushed_at.isoformat() if p.last_pushed_at else None,
        })
    if output_format == "json":
        click.echo(json.dumps({"workspace": ws.slug, "projects": rows}, indent=2))
        return
    click.echo(f"Workspace: {ws.name} ({ws.slug})")
    for r in rows:
        click.echo(f"  {r['slug']:12s} pulled {r['last_pulled_at'] or '—'}, "
                   f"pushed {r['last_pushed_at'] or '—'}")


def _status_project(proj_dir: Path, output_format: str) -> None:
    cfg = load_project_config(proj_dir)
    if cfg.workspace is None:
        click.echo("project is not linked to a workspace")
        return
    ws_resolved = (proj_dir / cfg.workspace.path).resolve()
    from keel.core.node_store import list_nodes
    nodes = list_nodes(proj_dir)
    workspace_origin = [n for n in nodes
                        if getattr(n, "origin", "local") == "workspace"]
    promotion_candidates = [n for n in nodes
                             if getattr(n, "origin", "local") == "local"
                             and getattr(n, "scope", "local") == "workspace"]
    forks = [n for n in nodes
             if getattr(n, "origin", "local") == "workspace"
             and getattr(n, "scope", "local") == "local"]

    if output_format == "json":
        click.echo(json.dumps({
            "workspace_path": str(ws_resolved),
            "workspace_origin_count": len(workspace_origin),
            "promotion_candidate_count": len(promotion_candidates),
            "fork_count": len(forks),
        }, indent=2))
        return

    click.echo(f"Project linked to: {ws_resolved}")
    click.echo(f"  workspace-origin nodes: {len(workspace_origin)}")
    click.echo(f"  promotion candidates:   {len(promotion_candidates)}")
    click.echo(f"  forks:                  {len(forks)}")

```python
"""Three-way merge engine for workspace-origin nodes.

Compares base (workspace at last_pulled_sha), ours (current project
copy), and theirs (current workspace HEAD). Returns a MergeResult
with a status and either a merged dict (trivial cases) or a
conflicting_fields list (non-trivial — handled by merge brief).
"""

from __future__ import annotations

from dataclasses import dataclass, field
from enum import Enum
from typing import Any


class MergeStatus(Enum):
    FAST_FORWARD = "fast_forward"          # ours == base, take theirs
    NO_UPSTREAM_CHANGES = "no_upstream"    # theirs == base, keep ours
    NO_CHANGES = "no_changes"               # ours == base == theirs
    AUTO_MERGED = "auto_merged"            # non-overlapping changes
    CONFLICT = "conflict"                   # overlapping changes, need agent


@dataclass
class MergeResult:
    status: MergeStatus
    merged: dict[str, Any] | None = None        # populated for non-CONFLICT
    conflicting_fields: list[str] = field(default_factory=list)
    auto_merged_fields: list[str] = field(default_factory=list)


# Fields we don't merge (bookkeeping, not semantic content):
_EXCLUDED_FROM_MERGE = {
    "uuid", "created_at", "updated_at",
    "origin", "scope", "workspace_sha", "workspace_pulled_at",
}


def merge_nodes(
    *, base: dict[str, Any], ours: dict[str, Any], theirs: dict[str, Any]
) -> MergeResult:
    """Three-way merge two node dicts against their common ancestor.

    Returns a MergeResult. On CONFLICT, callers should generate a merge
    brief from (base, ours, theirs) rather than using `merged`.
    """
    # Identify semantically-relevant fields.
    all_keys = (set(base) | set(ours) | set(theirs)) - _EXCLUDED_FROM_MERGE

    if _equal(ours, base):
        return MergeResult(
            status=(MergeStatus.NO_CHANGES if _equal(theirs, base)
                    else MergeStatus.FAST_FORWARD),
            merged=dict(theirs),
        )

    if _equal(theirs, base):
        return MergeResult(
            status=MergeStatus.NO_UPSTREAM_CHANGES,
            merged=dict(ours),
        )

    # Non-overlapping field merge.
    merged: dict[str, Any] = dict(base)
    conflicting: list[str] = []
    auto: list[str] = []
    for key in all_keys:
        b = base.get(key)
        o = ours.get(key)
        t = theirs.get(key)
        if o == t:
            merged[key] = o
            continue
        if o == b:
            merged[key] = t
            auto.append(key)
        elif t == b:
            merged[key] = o
            auto.append(key)
        else:
            conflicting.append(key)

    if conflicting:
        return MergeResult(
            status=MergeStatus.CONFLICT,
            conflicting_fields=conflicting,
            auto_merged_fields=auto,
        )
    return MergeResult(
        status=MergeStatus.AUTO_MERGED,
        merged=merged,
        auto_merged_fields=auto,
    )


def _equal(a: dict, b: dict) -> bool:
    """Compare dicts excluding bookkeeping fields."""
    ak = {k: v for k, v in a.items() if k not in _EXCLUDED_FROM_MERGE}
    bk = {k: v for k, v in b.items() if k not in _EXCLUDED_FROM_MERGE}
    return ak == bk
```

def workspace_push_cmd(

) -> None:
    """Propose project node changes upstream to workspace."""
    proj = project_dir.expanduser().resolve()
    _require_project(proj)
    cfg = load_project_config(proj)
    if cfg.workspace is None:
        raise click.ClickException("project is not linked to a workspace")
    ws_dir = (proj / cfg.workspace.path).resolve()

    # Collect push candidates: modified workspace-origin nodes and
    # local-origin scope=workspace nodes (promotion candidates).
    from keel.core.node_store import list_nodes, save_node
    from keel.core.workspace_sync import MergeStatus, merge_nodes
    from datetime import datetime, timezone

    target_ids = set(nodes.split(",")) if nodes else None
    pushes: list[tuple[str, str, dict]] = []  # (node_id, action, final_dict)
    conflicts: list[str] = []

    for node in list_nodes(proj):
        if target_ids is not None and node.id not in target_ids:
            continue
        origin = getattr(node, "origin", "local")
        scope = getattr(node, "scope", "local")
        if origin == "workspace" and scope == "workspace":
            # Modified upstream-origin. Check divergence.
            ours_dict = node.model_dump(mode="python")
            theirs_dict = _load_workspace_node(ws_dir, node.id).model_dump(mode="python")
            base_dict = _git_show_node(ws_dir, node.workspace_sha, node.id)
            result = merge_nodes(base=base_dict, ours=ours_dict, theirs=theirs_dict)
            if result.status is MergeStatus.CONFLICT:
                conflicts.append(node.id)
                continue
            if result.status is MergeStatus.NO_UPSTREAM_CHANGES:
                pushes.append((node.id, "fast-forward", ours_dict))
            elif result.status is MergeStatus.AUTO_MERGED:
                pushes.append((node.id, "auto-merged", result.merged))
            # NO_CHANGES / FAST_FORWARD: nothing to push.
        elif origin == "local" and scope == "workspace":
            # Promotion candidate. Check ID collision.
            from keel.core.paths import workspace_node_path
            if workspace_node_path(ws_dir, node.id).exists():
                conflicts.append(node.id)  # collision
                continue
            pushes.append((node.id, "promotion", node.model_dump(mode="python")))

    if conflicts:
        click.echo("Cannot push — upstream divergence or collision on:")
        for n in conflicts:
            click.echo(f"  {n}")
        click.echo("Run `keel workspace pull` first, or `keel workspace fork <id>`.")
        raise click.exceptions.Exit(11)

    if not pushes:
        click.echo("nothing to push")
        return

    if dry_run:
        for node_id, action, _ in pushes:
            click.echo(f"would {action}: {node_id}")
        return

    # Apply pushes to workspace repo, commit, update bookkeeping.
    from keel.core.paths import workspace_node_path
    from keel.core.parser import dump_frontmatter_yaml
    for node_id, _action, final_dict in pushes:
        # Strip project-only bookkeeping; store canonical form.
        canonical = dict(final_dict)
        canonical.update({
            "origin": "workspace", "scope": "workspace",
        })
        canonical.pop("workspace_sha", None)  # canonical nodes don't have it
        canonical.pop("workspace_pulled_at", None)

```python
"""Merge brief generator and store.

A merge brief is a structured YAML file written to
`<project>/.keel/merge-briefs/<node-id>.yaml` when a 3-way merge
surfaces non-trivial conflicts. The PM agent reads the brief, edits
the node file to a resolved form, and runs `keel workspace merge-resolve`
to finalize.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from pathlib import Path
from typing import Any, Literal

import yaml

from keel.core.paths import merge_brief_path, merge_briefs_dir


class MergeType(Enum):
    PULL = "pull"
    PUSH = "push"


@dataclass
class FieldDiff:
    field: str
    base: Any
    ours: Any
    theirs: Any
    status: Literal["ours_only", "theirs_only", "conflict", "structural"]


@dataclass
class MergeBrief:
    node_id: str
    merge_type: MergeType
    base_sha: str
    generated_at: datetime
    base_version: dict[str, Any]
    ours_version: dict[str, Any]
    theirs_version: dict[str, Any]
    field_diffs: list[FieldDiff]
    auto_merged_fields: list[str]
    hints: list[str] = field(default_factory=list)


_EXCLUDED = {
    "uuid", "created_at", "updated_at",
    "origin", "scope", "workspace_sha", "workspace_pulled_at",
}


def build_merge_brief(
    *, node_id: str, merge_type: MergeType, base_sha: str,
    base: dict, ours: dict, theirs: dict,
    auto_merged_fields: list[str] | None = None,
) -> MergeBrief:
    """Construct a MergeBrief from the three versions.

    Generates field_diffs for all fields that differ across the three
    versions, with status classifying each difference.
    """
    diffs: list[FieldDiff] = []
    all_keys = (set(base) | set(ours) | set(theirs)) - _EXCLUDED

    for key in sorted(all_keys):
        b = base.get(key)
        o = ours.get(key)
        t = theirs.get(key)
        if b == o == t:
            continue
        if o == b:
            status = "theirs_only"
        elif t == b:
            status = "ours_only"
        elif o == t:
            status = "structural"  # both changed to same new value
        else:
            status = "conflict"
        diffs.append(FieldDiff(field=key, base=b, ours=o, theirs=t, status=status))

    hints = _generate_hints(diffs)

    return MergeBrief(
        node_id=node_id,
        merge_type=merge_type,
        base_sha=base_sha,
        generated_at=datetime.now(tz=timezone.utc),
        base_version=base,
        ours_version=ours,
        theirs_version=theirs,
        field_diffs=diffs,
        auto_merged_fields=auto_merged_fields or [],
        hints=hints,
    )


def _generate_hints(diffs: list[FieldDiff]) -> list[str]:
    """Plain-language hints to help the agent reason about the merge."""
    hints: list[str] = []
    for d in diffs:
        if d.status != "conflict":
            continue
        if isinstance(d.ours, str) and isinstance(d.theirs, str):
            hints.append(
                f"field `{d.field}`: both sides modified the text. Consider whether "
                "the changes are additive (merge) or truly contradictory (pick one)."
            )
        elif isinstance(d.ours, list) and isinstance(d.theirs, list):
            hints.append(
                f"field `{d.field}`: both sides modified the list. Union may be "
                "the right answer if the additions are independent."
            )
    if not hints and diffs:
        hints.append("Non-overlapping field changes present — review each "
                     "and pick base/ours/theirs as appropriate.")
    return hints


def save_merge_brief(project_dir: Path, brief: MergeBrief) -> None:
    merge_briefs_dir(project_dir).mkdir(parents=True, exist_ok=True)
    data = {
        "node_id": brief.node_id,
        "merge_type": brief.merge_type.value,
        "base_sha": brief.base_sha,
        "generated_at": brief.generated_at.isoformat(),
        "base_version": brief.base_version,
        "ours_version": brief.ours_version,
        "theirs_version": brief.theirs_version,
        "field_diffs": [
            {"field": d.field, "base": d.base, "ours": d.ours,
             "theirs": d.theirs, "status": d.status}
            for d in brief.field_diffs
        ],
        "auto_merged_fields": brief.auto_merged_fields,
        "hints": brief.hints,
    }
    path = merge_brief_path(project_dir, brief.node_id)
    path.write_text(yaml.safe_dump(data, sort_keys=False))


def load_merge_brief(project_dir: Path, node_id: str) -> MergeBrief | None:
    path = merge_brief_path(project_dir, node_id)
    if not path.exists():
        return None
    data = yaml.safe_load(path.read_text())
    return MergeBrief(
        node_id=data["node_id"],
        merge_type=MergeType(data["merge_type"]),
        base_sha=data["base_sha"],
        generated_at=datetime.fromisoformat(data["generated_at"]),
        base_version=data["base_version"],
        ours_version=data["ours_version"],
        theirs_version=data["theirs_version"],
        field_diffs=[FieldDiff(**d) for d in data["field_diffs"]],
        auto_merged_fields=data.get("auto_merged_fields", []),
        hints=data.get("hints", []),
    )


def delete_merge_brief(project_dir: Path, node_id: str) -> None:
    path = merge_brief_path(project_dir, node_id)
    if path.exists():
        path.unlink()


def list_pending_briefs(project_dir: Path) -> list[str]:
    """Return node IDs that have pending merge briefs."""
    d = merge_briefs_dir(project_dir)
    if not d.exists():
        return []
    return sorted(p.stem for p in d.glob("*.yaml"))
```

```markdown
# Workspace sync — agent mediation guide

This doc explains how to handle a merge brief produced by
`keel workspace pull`. It's referenced by the `/pm-project-sync`
slash command.

## When this applies

A merge brief is created when `keel workspace pull` encounters a
non-trivial 3-way conflict on a workspace-origin node. Trivial cases
(fast-forward, non-overlapping field changes) are auto-applied; only
genuinely overlapping edits reach you.

## Brief structure

File: `.keel/merge-briefs/<node-id>.yaml`

```yaml
node_id: auth-system
merge_type: pull
base_sha: a3f2b1c                  # workspace commit at last pull
generated_at: 2026-04-15T14:30:00Z
base_version: {...}                # node at base_sha
ours_version: {...}                # project's current copy
theirs_version: {...}              # workspace HEAD
field_diffs:
  - field: description
    base: "..."
    ours: "..."
    theirs: "..."
    status: conflict
auto_merged_fields:
  - related_to                      # applied already in draft node
hints:
  - "field `description`: both sides modified the text..."
```

A draft-merged node file is also written to `nodes/<node-id>.yaml`.
It contains the auto-merged fields already applied plus your `ours`
values as placeholders for conflicting fields. Edit this file to
produce the final resolved node.

## Field diff statuses

| status | meaning | action |
|---|---|---|
| `ours_only` | Only we changed this field | Already taken; leave as-is |
| `theirs_only` | Only upstream changed this field | Already taken; leave as-is |
| `conflict` | Both sides changed it differently | Reason and resolve |
| `structural` | Both sides changed it to the same new value | Already consistent |

Only `conflict` entries need your intelligence.

## Resolution strategies

### Text fields (`description`, `definition`)

- **Both extended with compatible content**: combine. Example — base
  said "Auth service"; ours said "Auth service, handles sessions"; 
  theirs said "Auth service supporting OAuth". Resolve: "Auth service 
  supporting OAuth, handles sessions."
- **Contradictory meanings**: pick the one matching project purpose. 
  Note the deviation in the description if it's meaningful.
- **Upstream generic + local specific**: usually keep upstream's general
  definition and append a project-specific clarifier.

### List fields (`related_to`, `aliases`)

- Non-overlapping additions — union (`[a, b]` + `[a, c]` → `[a, b, c]`).
- Removed upstream but still relevant locally — keep as fork concern;
  consider `keel workspace fork` if this project really needs the old
  shape.

### Typed fields (`kind`, enum values)

- Genuinely different types are a hint that this might be two distinct
  concepts. Consider forking and creating a new node rather than
  reconciling.

## Finalizing

After editing `nodes/<node-id>.yaml` to the resolved form:

```bash
keel workspace merge-resolve <node-id>
```

This validates the node schema, bumps `workspace_sha` to the current
workspace HEAD, and deletes the brief. If validation fails, the brief
is preserved — fix the node and retry.

## When to fork instead of merge

- Your project needs a genuinely different version of this concept
- Upstream has evolved in a direction that doesn't fit your use case
- You want to stop receiving updates for this node entirely

Fork via `keel workspace fork <node-id>`. The node's `scope` flips to
`local`; sync operations skip it in both directions. The `workspace_sha`
is preserved for audit but is no longer checked.

## When to abandon the pull

If a pull produced briefs you don't want to resolve right now:

```bash
rm .keel/merge-briefs/<node-id>.yaml
# Then revert nodes/<node-id>.yaml to its pre-pull state (check git).
```

The project's `workspace_sha` for that node stays at the pre-pull
value. Next pull will surface the same conflict again.
```

"""lint/stale_workspace_nodes — warn when workspace-origin nodes are
behind workspace HEAD."""

import subprocess
from pathlib import Path

from keel.core.linter import LintFinding, register_rule
from keel.core.node_store import list_nodes
from keel.core.store import load_project_config


@register_rule(stage="scoping", code="lint/stale_workspace_nodes", severity="warning")
def _check(ctx):
    yield from _emit(ctx)


@register_rule(stage="handoff", code="lint/stale_workspace_nodes", severity="warning")
def _check_handoff(ctx):
    yield from _emit(ctx)


def _emit(ctx):
    cfg = load_project_config(ctx.project_dir)
    if cfg.workspace is None:
        return
    ws_dir = (ctx.project_dir / cfg.workspace.path).resolve()
    if not ws_dir.exists():
        return  # orphan pointer — handled by other rule/error
    head = subprocess.run(
        ["git", "rev-parse", "--short", "HEAD"],
        cwd=ws_dir, capture_output=True, text=True,
    ).stdout.strip()

    for n in list_nodes(ctx.project_dir):
        origin = getattr(n, "origin", "local")
        scope = getattr(n, "scope", "local")
        if origin != "workspace" or scope != "workspace":
            continue
        sha = getattr(n, "workspace_sha", None)
        if sha and sha != head:
            yield LintFinding(
                code="lint/stale_workspace_nodes",
                severity="warning",
                message=(
                    f"node {n.id}: workspace_sha {sha} behind HEAD {head}"
                ),
                file=f"nodes/{n.id}.yaml",
                fix_hint="Run /pm-project-sync or `keel workspace pull`.",
            )
